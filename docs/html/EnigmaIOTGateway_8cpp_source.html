328  * -------------------------------------------------------------------------
329  *| msgType (1) | IV (16) | random (31 bits) | SleepyNode (1 bit) | CRC (4) |
330  * -------------------------------------------------------------------------
331  */
332 
333  struct __attribute__ ((packed, aligned (1))) {
334  uint8_t msgType;
335  uint8_t iv[IV_LENGTH];

339 
340 #define KEFMSG_LEN sizeof(keyExchangeFinished_msg)
341 
342  uint32_t crc32;
343  bool sleepyNode;
344 
345  if (count < KEFMSG_LEN) {
346  DEBUG_WARN ("Wrong message");
347  return false;
348  }
349 
350  memcpy (&keyExchangeFinished_msg, buf, KEFMSG_LEN);
351 
352  Crypto.decryptBuffer (
353  (uint8_t *)&(keyExchangeFinished_msg.random),
354  (uint8_t *)&(keyExchangeFinished_msg.random),
355  RANDOM_LENGTH + CRC_LENGTH,
356  keyExchangeFinished_msg.iv,
357  IV_LENGTH,
358  node->getEncriptionKey (),
359  KEY_LENGTH
360  );
361 
362  DEBUG_VERBOSE ("Decripted Key Exchange Finished message: %s", printHexBuffer ((uint8_t *)&keyExchangeFinished_msg, KEFMSG_LEN));
363 
364  memcpy (&crc32, &(keyExchangeFinished_msg.crc), CRC_LENGTH);
365 
366  if (!checkCRC ((uint8_t *)&keyExchangeFinished_msg, KEFMSG_LEN - CRC_LENGTH, &crc32)) {
367  DEBUG_WARN ("Wrong CRC");
368  return false;
369  }

375
