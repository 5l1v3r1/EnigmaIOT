632 bool EnigmaIOTSensorClass::dataMessage (const uint8_t *data, size_t len, bool controlMessage) {
633  /*
634  * --------------------------------------------------------------------------------------
635  *| msgType (1) | IV (16) | length (2) | NodeId (2) | Counter (2) | Data (....) | CRC (4) |
636  * --------------------------------------------------------------------------------------
637  */
638 
639  uint8_t buffer[200];
640  uint32_t crc32;
641  uint32_t counter;
642  uint16_t nodeId = node.getNodeId ();
643 
644  uint8_t *msgType_p = buffer;
645  uint8_t *iv_p = buffer + 1;
646  uint8_t *length_p = buffer + 1 + IV_LENGTH;
647  uint8_t *nodeId_p = buffer + 1 + IV_LENGTH + sizeof (int16_t);
648  uint8_t *counter_p = buffer + 1 + IV_LENGTH + sizeof (int16_t) + sizeof (int16_t);
649  uint8_t *data_p = buffer + 1 + IV_LENGTH + sizeof (int16_t) + sizeof (int16_t) + sizeof (int16_t);
650 
651  if (!data) {
652  return false;
653  }
654 
655  if (controlMessage) {
656  *msgType_p = (uint8_t)CONTROL_DATA;
657  } else {
658  *msgType_p = (uint8_t)SENSOR_DATA;
659  }

662

664 
665  memcpy (nodeId_p, &nodeId, sizeof (uint16_t));
666 
667  if (!controlMessage) { // Control messages do not use counter
668  if (useCounter) {
669  counter = node.getLastMessageCounter () + 1;
670  node.setLastMessageCounter (counter);
671  rtcmem_data.lastMessageCounter = counter;
672  }
673  else {
674  counter = Crypto.random ();
675  }
676 
677  memcpy (counter_p, &counter, sizeof (uint16_t));
678  }
679 
680  memcpy (data_p, data, len);
681 
682  uint16_t packet_length = 1 + IV_LENGTH + sizeof (int16_t) + sizeof (int16_t) + sizeof (uint16_t) + len;
683

687  DEBUG_VERBOSE ("CRC32 = 0x%08X", crc32);
688 
689  // int is little indian mode on ESP platform
690  uint8_t *crc_p = (uint8_t*)(buffer + packet_length);
691 
692  memcpy (crc_p, &crc32, CRC_LENGTH);
693 
694  DEBUG_VERBOSE ("Data message: %s", printHexBuffer (buffer, packet_length + CRC_LENGTH));
695 
696  uint8_t *crypt_buf = buffer + 1 + IV_LENGTH;
697 
698  size_t cryptLen = packet_length + CRC_LENGTH - 1 - IV_LENGTH;
699 
700  Crypto.encryptBuffer (crypt_buf, crypt_buf, cryptLen, iv_p, IV_LENGTH, node.getEncriptionKey (), KEY_LENGTH);
701 
702  DEBUG_VERBOSE ("Encrypted data message: %s", printHexBuffer (buffer, packet_length + CRC_LENGTH));
703 
704  if (controlMessage) {
705  DEBUG_INFO (" -------> CONTROL MESSAGE");
706  } else {
707  DEBUG_INFO (" -------> DATA");
708  }
709 #if DEBUG_LEVEL >= VERBOSE
710  char macStr[18];
711  DEBUG_DBG ("Destination address: %s", mac2str (rtcmem_data.gateway, macStr));
712 #endif
713 
714  if (useCounter && !otaRunning) { // RTC must not be written if OTA is running. OTA uses RTC memmory to signal 2nd firmware boot
715  rtcmem_data.crc32 = CRC32::calculate ((uint8_t *)rtcmem_data.nodeKey, sizeof (rtcmem_data) - sizeof (uint32_t));
716  if (ESP.rtcUserMemoryWrite (0, (uint32_t*)&rtcmem_data, sizeof (rtcmem_data))) {
717  DEBUG_DBG ("Write configuration data to RTC memory");
718 #if DEBUG_LEVEL >= VERBOSE
719  DEBUG_VERBOSE ("Write RTCData: %s", printHexBuffer ((uint8_t *)&rtcmem_data, sizeof (rtcmem_data)));
720  dumpRtcData (&rtcmem_data);
721 #endif
722  }
723  }
724
