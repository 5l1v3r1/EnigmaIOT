322 bool EnigmaIOTSensorClass::dataMessage (const uint8_t *data, size_t len) {
323  /*
324  * --------------------------------------------------------------------------------------
325  *| msgType (1) | IV (16) | length (2) | NodeId (2) | Counter (2) | Data (....) | CRC (4) |
326  * --------------------------------------------------------------------------------------
327  */
328 
329  uint8_t buffer[200];
330  uint32_t crc32;
331  uint32_t counter;
332  uint16_t nodeId = node.getNodeId ();
333 
334  uint8_t *msgType_p = buffer;
335  uint8_t *iv_p = buffer + 1;
336  uint8_t *length_p = buffer + 1 + IV_LENGTH;
337  uint8_t *nodeId_p = buffer + 1 + IV_LENGTH + sizeof (int16_t);
338  uint8_t *counter_p = buffer + 1 + IV_LENGTH + sizeof (int16_t) + sizeof (int16_t);
339  uint8_t *data_p = buffer + 1 + IV_LENGTH + sizeof (int16_t) + sizeof (int16_t) + sizeof (int16_t);
340 
341  if (!data) {
342  return false;
343  }
344 
345  *msgType_p = (uint8_t)SENSOR_DATA;
346 
347  CryptModule::random (iv_p, IV_LENGTH);
348 
349  DEBUG_VERBOSE ("IV: %s", printHexBuffer (iv_p, IV_LENGTH));
350 
351  memcpy (nodeId_p, &nodeId, sizeof (uint16_t));
352 
353  if (useCounter) {
354  counter = node.getLastMessageCounter() + 1;
355  node.setLastMessageCounter (counter);
356  rtcmem_data.lastMessageCounter = counter;
357  }
358  else {
359  counter = Crypto.random ();
360  }
361 
362  memcpy (counter_p, &counter, sizeof(uint16_t));
363 
364  memcpy (data_p, data, len);
365 
366  uint16_t packet_length = 1 + IV_LENGTH + sizeof (int16_t) + sizeof (int16_t) + sizeof (uint16_t) + len;
367 
368  memcpy (length_p, &packet_length, sizeof (uint16_t));
369 
370  crc32 = CRC32::calculate (buffer, packet_length);
371  DEBUG_VERBOSE ("CRC32 = 0x%08X", crc32);
372 
373  // int is little indian mode on ESP platform
374  uint8_t *crc_p = (uint8_t*)(buffer + packet_length);
375 
376  memcpy (crc_p, &crc32, CRC_LENGTH);
377 
378  DEBUG_VERBOSE ("Data message: %s", printHexBuffer (buffer, packet_length + CRC_LENGTH));
379 
380  uint8_t *crypt_buf = buffer + 1 + IV_LENGTH;
381 
382  size_t cryptLen = packet_length + CRC_LENGTH - 1 - IV_LENGTH;
383 
384  Crypto.encryptBuffer (crypt_buf, crypt_buf, cryptLen, iv_p, IV_LENGTH, node.getEncriptionKey (), KEY_LENGTH);
385 
386  DEBUG_VERBOSE ("Encrypted data message: %s", printHexBuffer (buffer, packet_length + CRC_LENGTH));
387 
388  DEBUG_INFO (" -------> DATA");
389 
390  if (useCounter) {
391  rtcmem_data.crc32 = CRC32::calculate ((uint8_t *)rtcmem_data.nodeKey, sizeof (rtcmem_data) - sizeof (uint32_t));
392  if (ESP.rtcUserMemoryWrite (0, (uint32_t*)&rtcmem_data, sizeof (rtcmem_data))) {
393  DEBUG_VERBOSE ("Write RTCData: %s", printHexBuffer ((uint8_t *)&rtcmem_data, sizeof (rtcmem_data)));
394  }
395  }
396 
397  return comm->send (gateway, buffer, packet_length + CRC_LENGTH) == 0;
398 }

466 void EnigmaIOTSensorClass::manageMessage (const uint8_t *mac, const uint8_t* buf, uint8_t count) {
467  DEBUG_INFO ("Reveived message. Origin MAC: %02X:%02X:%02X:%02X:%02X:%02X", mac[0], mac[1], mac[2], mac[3], mac[4], mac[5]);
468  DEBUG_VERBOSE ("Received data: %s", printHexBuffer (const_cast<uint8_t *>(buf), count));
469  flashBlue = true;
470 
471  if (count <= 1) {
472  return;
473  }
474 
475  switch (buf[0]) {
476  case SERVER_HELLO:
477  DEBUG_INFO (" <------- SERVER HELLO");
478  if (node.getStatus () == WAIT_FOR_SERVER_HELLO) {
479  if (processServerHello (mac, buf, count)) {
480  keyExchangeFinished ();
481  node.setStatus (WAIT_FOR_CIPHER_FINISHED);
482  } else {
483  node.reset ();
484  }
485  } else {
486  node.reset ();
487  }
488  break;
489  case CYPHER_FINISHED:
490  DEBUG_INFO ("Cypher Finished received");
491  if (node.getStatus () == WAIT_FOR_CIPHER_FINISHED) {
492  DEBUG_INFO (" <------- CIPHER_FINISHED");
493  if (processCipherFinished (mac, buf, count)) {
494  // mark node as registered
495  node.setKeyValid (true);
496  node.setKeyValidFrom (millis ());
497  node.setLastMessageCounter (0);
498  node.setStatus (REGISTERED);
499 
500  memcpy (rtcmem_data.nodeKey, node.getEncriptionKey (), KEY_LENGTH);
501  rtcmem_data.lastMessageCounter = 0;
502  rtcmem_data.nodeId = node.getNodeId ();
503  rtcmem_data.crc32 = CRC32::calculate ((uint8_t *)rtcmem_data.nodeKey, sizeof (rtcmem_data) - sizeof (uint32_t));
504  if (ESP.rtcUserMemoryWrite (0, (uint32_t*)&rtcmem_data, sizeof (rtcmem_data))) {
505  DEBUG_VERBOSE ("Write RTCData: %s", printHexBuffer ((uint8_t *)&rtcmem_data, sizeof (rtcmem_data)));
506  }
507 
508 #if DEBUG_LEVEL >= INFO
509  node.printToSerial (&DEBUG_ESP_PORT);
510 #endif
511  if (notifyConnection) {
512  notifyConnection ();
513  }
514  // Resend last message in case of it is still pending to be sent.
515  // If key expired it was successfully sent before so retransmission is not needed
516  if (invalidateReason < KEY_EXPIRED && dataMessageSentLength > 0) {
517  if (node.getStatus () == REGISTERED && node.isKeyValid ()) {
518  DEBUG_INFO ("Data sent: %s", printHexBuffer (dataMessageSent, dataMessageSentLength));
519  dataMessage ((uint8_t *)dataMessageSent, dataMessageSentLength);
520  flashBlue = true;
521  }
522  }
523  // TODO: Store node data on EEPROM, SPIFFS or RTCMEM
524 
525  } else {
526  node.reset ();
527  }
528  } else {
529  node.reset ();
530  }
531  break;
532  case INVALIDATE_KEY:
533  DEBUG_INFO (" <------- INVALIDATE KEY");
534  invalidateReason = processInvalidateKey (mac, buf, count);
535  node.reset ();
536  if (notifyDisconnection) {
537  notifyDisconnection ();
538  }
539  break;
540  case DOWNSTREAM_DATA:
541  DEBUG_INFO (" <------- DOWNSTREAM DATA");
542  if (processDownstreamData (mac, buf, count)) {
543  DEBUG_INFO ("Downstream Data OK");
544  }
545  break;
546  }
547 }
